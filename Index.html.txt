    <!-- Firebase SDKs for authentication and database -->
    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-app.js";
        import { getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-auth.js";
        import { getFirestore, doc, setDoc, collection, getDocs, onSnapshot, query, where, addDoc } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-firestore.js";
        import { setLogLevel } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-app.js";

        // Set Firebase debug logs
        setLogLevel('debug');
        
        // Use global variables provided by the environment
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';
        const firebaseConfig = JSON.parse(typeof __firebase_config !== 'undefined' ? __firebase_config : '{}');

        // Initialize Firebase
        const app = initializeApp(firebaseConfig);
        const db = getFirestore(app);
        const auth = getAuth(app);
        
        let userId = '';
        let clientDetails = {};

        // UI elements
        const step1 = document.getElementById('step1');
        const step2 = document.getElementById('step2');
        const step3 = document.getElementById('step3');

        const clientForm = document.getElementById('client-form');
        const clientNameInput = document.getElementById('clientName');
        const clientEmailInput = document.getElementById('clientEmail');
        const clientPhoneInput = document.getElementById('clientPhone');

        const calendarContainer = document.getElementById('calendar-container');
        const calendarGrid = document.querySelector('.calendar-grid');
        const currentMonthDisplay = document.getElementById('currentMonth');
        const prevMonthBtn = document.getElementById('prevMonth');
        const nextMonthBtn = document.getElementById('nextMonth');
        const timeSlotsContainer = document.getElementById('time-slots-container');
        const timeSlotsGrid = document.getElementById('time-slots-grid');
        const selectedDateHeading = document.getElementById('selected-date-heading');

        const whatsappConfirmBtn = document.getElementById('whatsapp-confirm-btn');
        const bookedDateSpan = document.getElementById('booked-date');
        const bookedTimeSpan = document.getElementById('booked-time');

        let currentDate = new Date();
        const bookings = {};
        const availableTimes = ['09:00 AM', '10:00 AM', '11:00 AM', '12:00 PM', '01:00 PM', '02:00 PM', '03:00 PM', '04:00 PM'];

        // Get user ID on auth state change
        onAuthStateChanged(auth, async (user) => {
            if (user) {
                userId = user.uid;
                await fetchBookings();
            } else {
                if (typeof __initial_auth_token !== 'undefined') {
                    await signInWithCustomToken(auth, __initial_auth_token);
                } else {
                    await signInAnonymously(auth);
                }
            }
        });

        const fetchBookings = async () => {
            const bookingsRef = collection(db, `/artifacts/${appId}/public/data/bookings`);
            const q = query(bookingsRef);
            
            onSnapshot(q, (snapshot) => {
                snapshot.docChanges().forEach((change) => {
                    const bookingData = change.doc.data();
                    const bookingDate = bookingData.date;
                    const bookingTime = bookingData.time;
                    
                    if (change.type === 'added') {
                        if (!bookings[bookingDate]) {
                            bookings[bookingDate] = new Set();
                        }
                        bookings[bookingDate].add(bookingTime);
                    }
                    if (change.type === 'modified') {
                        // Handle modified bookings if needed
                    }
                    if (change.type === 'removed') {
                        if (bookings[bookingDate]) {
                            bookings[bookingDate].delete(bookingTime);
                        }
                    }
                });
                renderCalendar(currentDate);
            });
        };

        const renderCalendar = (date) => {
            const year = date.getFullYear();
            const month = date.getMonth();
            const today = new Date();
            today.setHours(0, 0, 0, 0);

            const firstDayOfMonth = new Date(year, month, 1);
            const lastDayOfMonth = new Date(year, month + 1, 0);

            currentMonthDisplay.textContent = new Intl.DateTimeFormat('en-US', { month: 'long', year: 'numeric' }).format(date);
            
            const existingDays = document.querySelectorAll('.calendar-grid .calendar-cell:not(.header)');
            existingDays.forEach(cell => cell.remove());

            const startDayOfWeek = firstDayOfMonth.getDay();
            
            for (let i = 0; i < startDayOfWeek; i++) {
                const emptyCell = document.createElement('div');
                emptyCell.classList.add('calendar-cell');
                calendarGrid.appendChild(emptyCell);
            }

            for (let i = 1; i <= lastDayOfMonth.getDate(); i++) {
                const day = new Date(year, month, i);
                const dayCell = document.createElement('div');
                dayCell.classList.add('calendar-cell');
                dayCell.textContent = i;
                dayCell.dataset.date = day.toISOString().split('T')[0];
                
                if (day.getDay() !== 0 && day.getDay() !== 6 && day >= today) { 
                    const dayBookings = bookings[dayCell.dataset.date] || new Set();
                    if (dayBookings.size < availableTimes.length) {
                        dayCell.classList.add('available');
                        dayCell.addEventListener('click', () => selectDate(dayCell));
                    } else {
                        dayCell.classList.add('busy');
                    }
                } else {
                    dayCell.classList.add('busy');
                }

                if (day.getTime() === today.getTime()) {
                    dayCell.classList.add('today');
                }

                calendarGrid.appendChild(dayCell);
            }
        };

        let selectedDateCell = null;
        const selectDate = (cell) => {
            if (selectedDateCell) {
                selectedDateCell.classList.remove('selected');
            }
            selectedDateCell = cell;
            selectedDateCell.classList.add('selected');
            
            const dateStr = cell.dataset.date;
            displayTimeSlots(dateStr);
            
            // Scroll to the time slots container
            timeSlotsContainer.scrollIntoView({ behavior: 'smooth' });
        };

        const displayTimeSlots = (dateStr) => {
            timeSlotsGrid.innerHTML = '';
            timeSlotsContainer.classList.remove('hidden');
            
            selectedDateHeading.textContent = new Date(dateStr + 'T12:00:00').toLocaleDateString('en-US', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' });
            
            const bookedSlots = bookings[dateStr] || new Set();
            let availableCount = 0;

            availableTimes.forEach(time => {
                const timeSlot = document.createElement('div');
                timeSlot.classList.add('time-slot');
                timeSlot.textContent = time;

                if (bookedSlots.has(time)) {
                    timeSlot.classList.add('booked');
                } else {
                    timeSlot.addEventListener('click', () => bookAppointment(dateStr, time));
                    availableCount++;
                }
                timeSlotsGrid.appendChild(timeSlot);
            });

            if (availableCount === 0) {
                document.getElementById('no-slots-message').classList.remove('hidden');
            } else {
                document.getElementById('no-slots-message').classList.add('hidden');
            }
        };

        const bookAppointment = async (date, time) => {
            const bookingData = {
                date: date,
                time: time,
                clientName: clientDetails.name,
                clientEmail: clientDetails.email,
                clientPhone: clientDetails.phone,
                userId: userId,
                bookedAt: new Date().toISOString(),
            };
            
            try {
                const bookingsCollectionRef = collection(db, `/artifacts/${appId}/public/data/bookings`);
                await addDoc(bookingsCollectionRef, bookingData);
                showConfirmation(date, time);
            } catch (error) {
                console.error("Error adding document: ", error);
                const errorMessage = document.createElement('div');
                errorMessage.textContent = 'Error booking appointment. Please try again.';
                errorMessage.style.cssText = 'position:fixed;top:50%;left:50%;transform:translate(-50%,-50%);background:red;color:white;padding:20px;border-radius:10px;z-index:1000;';
                document.body.appendChild(errorMessage);
                setTimeout(() => errorMessage.remove(), 3000);
            }
        };

        const showConfirmation = (date, time) => {
            step2.classList.add('hidden');
            step3.classList.remove('hidden');
            
            bookedDateSpan.textContent = new Date(date + 'T12:00:00').toLocaleDateString('en-US', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' });
            bookedTimeSpan.textContent = time;

            const whatsappMessage = `Hi The Nail Bar! I'd like to confirm my booking.
Name: ${clientDetails.name}
Email: ${clientDetails.email}
Phone: ${clientDetails.phone}
Date: ${bookedDateSpan.textContent}
Time: ${bookedTimeSpan.textContent}`;
            
            const whatsappUrl = `https://wa.me/27817719639?text=${encodeURIComponent(whatsappMessage)}`;
            whatsappConfirmBtn.href = whatsappUrl;
            
            // Scroll to the confirmation message
            step3.scrollIntoView({ behavior: 'smooth' });
        };

        // Event Listeners for multi-step form
        clientForm.addEventListener('submit', (e) => {
            e.preventDefault();
            clientDetails.name = clientNameInput.value;
            clientDetails.email = clientEmailInput.value;
            clientDetails.phone = clientPhoneInput.value;
            
            step1.classList.add('hidden');
            step2.classList.remove('hidden');
            renderCalendar(currentDate);
            
            // Scroll to the next step
            step2.scrollIntoView({ behavior: 'smooth' });
        });

        prevMonthBtn.addEventListener('click', () => {
            currentDate.setMonth(currentDate.getMonth() - 1);
            renderCalendar(currentDate);
        });

        nextMonthBtn.addEventListener('click', () => {
            currentDate.setMonth(currentDate.getMonth() + 1);
            renderCalendar(currentDate);
        });
    </script>
